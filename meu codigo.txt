exercio praticar

1. Positivo ou negativo
n = int(input("Digite um número: "))
if n >= 0:
    print("Positivo")
else:
    print("Negativo")


 2. Dobro do valor
n = int(input("Digite um número: "))
print("Dobro:", n * 2)


 3. Mostrar valor e tipo
valor = input("Digite algo: ")
print("Valor digitado:", valor)
print("Tipo:", type(valor))


 4. Par ou ímpar
n = int(input("Digite um número: "))
if n % 2 == 0:
    print("Par")
else:
    print("Ímpar")


 5. Triplo do número (lido como texto)
texto = input("Digite um número: ")
n = int(texto)
print("Triplo:", n * 3)


 6. Maior entre dois números
a = int(input("Digite o primeiro número: "))
b = int(input("Digite o segundo número: "))
if a > b:
    print("Maior:", a)
else:
    print("Maior:", b)


 7. Maior que 10 ou não
n = float(input("Digite um valor: "))
if n > 10:
    print("Maior que 10")
else:
    print("Menor ou igual a 10")


 8. Raiz se positivo
n = float(input("Digite um número: "))
if n > 0:
    print("Raiz aproximada:", n ** 0.5)
else:
    print("Número inválido")


 9. Metade do valor (float)
valor = float(input("Digite um valor: "))
print("Metade:", valor / 2)


 10. Intervalo entre 0 e 10
n = float(input("Digite um número: "))

    print("Fora do intervalo")
# 11. Par positivo, par negativo ou ímpar
n = int(input("Digite um número: "))
if n % 2 == 0 and n > 0:
    print("Par positivo")
elif n % 2 == 0 and n < 0:
    print("Par negativo")
else:
    print("Ímpar")


12. Soma e maior ou iguais
a = float(input("Digite o primeiro número: "))
b = float(input("Digite o segundo número: "))
print("Soma:", a + b)

if a > b:
    print("Maior:", a)
elif b > a:
    print("Maior:", b)
else:
    print("São iguais")


 13. Maior que 100 → metade; senão → dobro
n = float(input("Digite um número: "))
if n > 100:
    print("Metade:", n / 2)
else:
    print("Dobro:", n * 2)

 15. Entre 10 e 20
n = float(input("Digite um número: "))

 16. Tipo e quadrado se numérico
valor = input("Digite algo: ")

 17. Classificação por idade
idade = int(input("Digite a idade: "))
if idade < 18:
    print("Menor de idade")
elif idade <= 59:
    print("Adulto")
else:
    print("Idoso")


 18. Classificação completa
n = int(input("Digite um número: "))
if n == 0:
    print("Neutro")
elif n % 2 == 0 and n > 0:
    print("Par positivo")
elif n % 2 == 0 and n < 0:
    print("Par negativo")
elif n % 2 != 0 and n > 0:
    print("Ímpar positivo")
else:
    print("Ímpar negativo")


 19. Iguais ou diferentes + diferença
a = float(input("Digite o primeiro número: "))
b = float(input("Digite o segundo número: "))
if a == b:
    print("São iguais")
else:
    print("São diferentes")
    print("Diferença:", abs(a - b))

 21. Múltiplo de 2 e 3 sendo positivo
n = int(input("Digite um número inteiro: "))

if n > 0:
    if n % 2 == 0 and n % 3 == 0:
        print("Múltiplo de 2 e 3")
    else:
        print("Não atende")
else:
    print("Número inválido")


 22. Conversão para inteiro com validação
valor = input("Digite um valor: ")

try:
    n = int(valor)
    if n > 10:
        print("Alto")
    else:
        print("Baixo")
except:
    print("Entrada inválida")


 23. Restos da divisão por 3 em ordem decrescente
a = int(input("Digite o primeiro número: "))
b = int(input("Digite o segundo número: "))
c = int(input("Digite o terceiro número: "))

restos = [a % 3, b % 3, c % 3]
restos.sort(reverse=True)

print("Restos em ordem decrescente:", restos)


 24. Intervalo de 1 a 100 + par ou ímpar
n = int(input("Digite um número: "))

if 1 <= n <= 100:
    if n % 2 == 0:
        print("Par no intervalo")
    else:
        print("Ímpar no intervalo")
else:
    print("Fora do intervalo")
